You can override service config, add containers, inject env vars, etc...


services:
  appserver:
    overrides:
      services:
        environment:
            BEHAT_PARAMS: >-
              {
                  "extensions" : {
                      "Behat\\MinkExtension" : {
                          "base_url" : "http://nginx/"
                        }
                    }
                }

You can add custom tooling commands, this is super flexible


tooling:
  test:
    service: appserver
    cmd:
      - "robo"
      - "test"
